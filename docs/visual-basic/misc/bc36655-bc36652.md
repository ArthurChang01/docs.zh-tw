 <span data-ttu-id="fa9b7-108">下列程式碼示範這個錯誤。</span><span class="sxs-lookup"><span data-stu-id="fa9b7-108">The following code demonstrates the error.</span></span>

    Sub Main()  
  
        Dim caller As New Class1  
        '' Not valid.  
        'caller.targetExtension(1, "2")

    Sub targetExtension(Of T)(ByVal p0 As Class1, ByVal p1 As T, ByVal p2 As T)  
    End Sub  
  
    Class Class1  
    End Class
