## <a name="example"></a><span data-ttu-id="8a50d-107">範例</span><span class="sxs-lookup"><span data-stu-id="8a50d-107">Example</span></span>

```  
// CS3010.cs  
  
using System;  
  
[assembly:CLSCompliant(true)]  
public interface I  
{  
    [CLSCompliant(false)]  
    int M();   // CS3010  
}

{  
    public int M()  
    {  
        return 1;  
    }  
  
    public static void Main()  
    {  
    }  
}  
```
