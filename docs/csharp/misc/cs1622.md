 <span data-ttu-id="821e1-107">下列範例會產生 CS1622：</span><span class="sxs-lookup"><span data-stu-id="821e1-107">The following sample generates CS1622:</span></span>  
  
```  
// CS1622.cs  
// compile with: /target:library  
using System.Collections;  
  
class C : IEnumerable  
{  
   public IEnumerator GetEnumerator()  
   {  
      return (IEnumerator) this;  // CS1622  
      yield return this;   // OK  
   }  
}  
```
