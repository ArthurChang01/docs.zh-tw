```csharp  
// CS0652.cs

public class Class1  
{  
   private static byte i = 0;  
   public static void Main()  
   {  
      short j = 256;  
      if (i == 256)   // CS0652, 256 is out of range for byte  
         i = 0;  
   }  
}  
```
