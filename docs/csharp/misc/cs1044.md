 <span data-ttu-id="edabe-104">編譯器發現無效的陳述式。</span><span class="sxs-lookup"><span data-stu-id="edabe-104">The compiler found an invalid statement.</span></span>

```  
// CS1044.cs  
using System;

{  
   public void Dispose()  
   {  
      Console.WriteLine("Res1.Dispose()");  
   }  
  
   public static void Main()

      using (MyClass mc1 = new MyClass(),  
             MyClass mc2 = new MyClass())   // CS1044, remove an instantiation  
      {  
      }  
   }  
}  
```
