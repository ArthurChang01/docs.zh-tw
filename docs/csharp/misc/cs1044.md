 <span data-ttu-id="f63cf-104">編譯器發現無效的陳述式。</span><span class="sxs-lookup"><span data-stu-id="f63cf-104">The compiler found an invalid statement.</span></span>

// CS1044.cs

{  
   public void Dispose()  
   {  
      Console.WriteLine("Res1.Dispose()");

      }  
   }  
}
