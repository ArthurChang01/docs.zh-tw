 <span data-ttu-id="60014-106">下列範例會產生 CS0245：</span><span class="sxs-lookup"><span data-stu-id="60014-106">The following sample generates CS0245:</span></span>  
  
```csharp  
// CS0245.cs  
using System;  
using System.Collections;  
  
class MyClass // : IDisposable  
{  
   /*  
   public void Dispose()  
   {  
      // cleanup code goes here  
   }  
   */  
  
   void m()  
   {  
      this.Finalize();   // CS0245  
      // this.Dispose();  
   }  
  
   public static void Main()  
   {  
   }  
}  
```
