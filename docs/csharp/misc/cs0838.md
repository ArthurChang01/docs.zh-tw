## <a name="example"></a><span data-ttu-id="4fb49-107">範例</span><span class="sxs-lookup"><span data-stu-id="4fb49-107">Example</span></span>  
 <span data-ttu-id="4fb49-108">下列範例會產生 CS0838：</span><span class="sxs-lookup"><span data-stu-id="4fb49-108">The following example generates CS0838:</span></span>  
  
```  
// cs0838.cs  
using System;  
using System.Linq;  
using System.Linq.Expressions;  
  
namespace TestNamespace  
{  
    class Test  
    {  
        static int Main()  
        {  
  
            Expression<Func<int[,]>> expr =  
                () => new int[2, 2] { { 1, 2 }, { 3, 4 } }; // CS0838  
  
            // try the following 2 lines instead  
            int[,] nums = new int[2, 2] { { 1, 2 }, { 3, 4 } };  
            Expression<Func<int[,]>> expr2 = () => nums;   
  
            return 1;  
        }  
    }  
}  
```
