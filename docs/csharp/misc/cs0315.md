## <a name="example"></a><span data-ttu-id="e9ec7-108">範例</span><span class="sxs-lookup"><span data-stu-id="e9ec7-108">Example</span></span>

// cs0315.cs  
public class ClassConstraint { }  
public struct ViolateClassConstraint { }  
  
public class Gen<T> where T : ClassConstraint  
{         
}

{

    {

    }  
}
