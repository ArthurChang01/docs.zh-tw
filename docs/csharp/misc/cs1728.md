## <a name="example"></a><span data-ttu-id="3d342-105">範例</span><span class="sxs-lookup"><span data-stu-id="3d342-105">Example</span></span>  
 <span data-ttu-id="3d342-106">下列範例會產生 CS1728：</span><span class="sxs-lookup"><span data-stu-id="3d342-106">The following sample generates CS1728:</span></span>  
  
```csharp  
// CS1728.cs  
class Test  
{  
   delegate T GetT<T>();  
   delegate T GetT1<T>(T t);  
  
   delegate bool E(object o);  
   delegate int I();  
   delegate string S();  
  
   static void Main()  
   {  
      int? x = null;  
      int? y = 5;  
  
      GetT<int> d1 = x.GetValueOrDefault;   // CS1728  
      GetT<int> d2 = y.GetValueOrDefault;   // CS1728  
      GetT1<int> d3 = x.GetValueOrDefault;   // CS1728  
      GetT1<int> d4 = y.GetValueOrDefault;   // CS1728  
   }  
}  
```
