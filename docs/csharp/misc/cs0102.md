## <a name="example"></a><span data-ttu-id="31491-106">範例</span><span class="sxs-lookup"><span data-stu-id="31491-106">Example</span></span>

```csharp  
{  
   public class MyClass  
   {  
      string s = "Hello";  
      string s = "Goodbye";   // CS0102  
  
      public void GetString()  
      {  
         string s = "Hello again";   // method scope, no error  
      }  
   }  
}  
```
