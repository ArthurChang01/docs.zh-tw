```csharp  
// CS0161.cs  
public class Test

   public static int Main() // CS0161  
   {  
      int i = 10;  
      if (i < 10)  
      {  
         return i;  
      }  
      else  
      {  
         // uncomment the following line to resolve  
         // return 1;  
      }  
   }  
}
```
